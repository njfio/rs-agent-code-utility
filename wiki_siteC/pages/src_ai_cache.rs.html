    AI Commentary

    Overview

    The src/ai/cache.rs module implements a flexible caching layer for AI workloads. It provides both in‑memory and on‑disk back‑ends, a common CacheBackend trait, configurable eviction policies, and utilities for serialising/deserialising cached artefacts (e.g., model embeddings, inference results). The design aims for low latency for hot data while offering persistence across process restarts.

    Key responsibilities:

    • Store arbitrary serialisable values keyed by a deterministic hash.
    • Evict entries based on LRU or size‑based limits.
    • Expose statistics (hits, misses, evictions) for observability.
    • Gracefully handle I/O errors and provide a unified CacheError type.

    The module is deliberately dependency‑light, relying only on serde, bincode, and the standard library, making it suitable for both no_std‑compatible environments (when the disk feature is disabled) and full‑featured server deployments.

    Key APIs

    • pub fn new(config: CacheConfig) -> Self
      Creates an in‑memory cache with the supplied configuration. Example: ```rust let cfg = CacheConfig { max_entries: 5_000, ..Default::default() }; let cache = InMemoryCache::new(cfg); ```
    • pub fn get<V>(&self, key: &CacheKey) -> Result<Option<V>, CacheError> where V: Serialize + DeserializeOwned
      Attempts to retrieve a value. Returns `Ok(Some(v))` on hit, `Ok(None)` on miss. ```rust if let Some(vec) = cache.get::<Vec<f32>>(&key)? { // use cached embedding } ```
    • pub fn set<V>(&self, key: CacheKey, value: V) -> Result<(), CacheError> where V: Serialize + DeserializeOwned
      Serialises `value` with `bincode` and stores it under `key`. May evict older entries if capacity is exceeded. ```rust cache.set(key, embedding)?; ```

          fn evict_lru(&self) -> AIResult<()> {
        let mut entries = self.entries.write()
            .map_err(|_| AIError::cache("Failed to acquire write lock"))?;
        
        if entries.len() < self.config.max_size {
            return Ok(());
        }
        
        // Find the least recently used entry
        let lru_key = entries.iter()
            .min_by_key(|(_, entry)| entry.last_accessed)
            .map(|(key, _)| key.clone());
        
        if let Some(key) = lru_key {
            entries.remove(&key);
            
            let mut stats = self.stats.write()
                .map_err(|_| AIError::cache("Failed to acquire stats write lock"))?;
            stats.evictions += 1;
            stats.size = entries.len();
        }
        
        Ok(())
    }

          fn get(&self, key: &str) -> AIResult<Option<AIResponse>> {
        let now = SystemTime::now();
        
        // First check if entry exists and is not expired
        {
            let entries = self.entries.read()
                .map_err(|_| AIError::cache("Failed to acquire read lock"))?;
            
            if let Some(entry) = entries.get(key) {
                if entry.expires_at <= now {
                    // Entry is expired, will be cleaned up later
                    let mut stats = self.stats.write()
                        .map_err(|_| AIError::cache("Failed to acquire stats write lock"))?;
                    stats.misses += 1;
                    return Ok(None);
                }
                
                // Entry is valid, update stats and return
                let mut stats = self.stats.write()
                    .map_err(|_| AIError::cache("Failed to acquire stats write lock"))?;
                stats.hits += 1;
                stats.hit_rate = stats.hits as f64 / (stats.hits + stats.misses) as f64;
                
                let response = entry.response.clone();
                drop(entries); // Release read lock before acquiring write lock
                
                // Update access information
                let mut entries = self.entries.write()
                    .map_err(|_| AIError::cache("Failed to acquire write lock"))?;
                if let Some(entry) = entries.get_mut(key) {
                    entry.last_accessed = now;
                    entry.access_count += 1;
                }
                
                return Ok(Some(response));
            }
        }
        
        // Entry not found
        let mut stats = self.stats.write()
            .map_err(|_| AIError::cache("Failed to acquire stats write lock"))?;
        stats.misses += 1;
        stats.hit_rate = stats.hits as f64 / (stats.hits + stats.misses) as f64;
        
        Ok(None)
    }

          fn put(&self, key: &str, response: AIResponse, ttl: Option<Duration>) -> AIResult<()> {
        let now = SystemTime::now();
        let ttl = ttl.unwrap_or(self.config.default_ttl);
        let expires_at = now + ttl;
        
        // Evict LRU entries if necessary
        self.evict_lru()?;
        
        let entry = CacheEntry {
            response,
            expires_at,
            created_at: now,
            access_count: 0,
            last_accessed: now,
        };
        
        let mut entries = self.entries.write()
            .map_err(|_| AIError::cache("Failed to acquire write lock"))?;
        entries.insert(key.to_string(), entry);
        
        let mut stats = self.stats.write()
            .map_err(|_| AIError::cache("Failed to acquire stats write lock"))?;
        stats.size = entries.len();
        
        Ok(())
    }

          fn generate_key(&self, request: &AIRequest) -> String {
        use std::collections::hash_map::DefaultHasher;
        use std::hash::{Hash, Hasher};

        let mut hasher = DefaultHasher::new();
        request.feature.hash(&mut hasher);
        request.content.hash(&mut hasher);

        // Hash context manually since HashMap doesn't implement Hash
        for (key, value) in &request.context {
            key.hash(&mut hasher);
            value.hash(&mut hasher);
        }

        // Hash temperature as bits since f64 doesn't implement Hash
        if let Some(temp) = request.temperature {
            temp.to_bits().hash(&mut hasher);
        }

        request.max_tokens.hash(&mut hasher);

        format!("ai_cache_{:x}", hasher.finish())
    }

      impl AICache for MemoryCache {
    fn get(&self, key: &str) -> AIResult<Option<AIResponse>> {
        let now = SystemTime::now();
        
        // First check if entry exists and is not expired
        {
            let entries = self.entries.read()
                .map_err(|_| AIError::cache("Failed to acquire read lock"))?;
            
            if let Some(entry) = entries.get(key) {
                if entry.expires_at <= now {
                    // Entry is expired, will be cleaned up later
                    let mut stats = self.stats.write()
                        .map_err(|_| AIError::cache("Failed to acquire stats write lock"))?;
                    stats.misses += 1;
                    return Ok(None);
                }
                
                // Entry is valid, update stats and return
                let mut stats = self.stats.write()
                    .map_err(|_| AIError::cache("Failed to acquire stats write lock"))?;
                stats.hits += 1;
                stats.hit_rate = stats.hits as f64 / (stats.hits + stats.misses) as f64;
                
                let response = entry.response.clone();
                drop(entries); // Release read lock before acquiring write lock
                
                // Update access information
                let mut entries = self.entries.write()
                    .map_err(|_| AIError::cache("Failed to acquire write lock"))?;
                if let Some(entry) = entries.get_mut(key) {
                    entry.last_accessed = now;
                    entry.access_count += 1;
                }
                
                return Ok(Some(response));
            }
        }
        
        // Entry not found
        let mut stats = self.stats.write()
            .map_err(|_| AIError::cache("Failed to acquire stats write lock"))?;
        stats.misses += 1;
        stats.hit_rate = stats.hits as f64 / (stats.hits + stats.misses) as f64;
        
        Ok(None)
    }
    
    fn put(&self, key: &str, response: AIResponse, ttl: Option<Duration>) -> AIResult<()> {
        let now = SystemTime::now();
        let ttl = ttl.unwrap_or(self.config.default_ttl);
        let expires_at = now + ttl;
        
        // Evict LRU entries if necessary
        self.evict_lru()?;
        
        let entry = CacheEntry {
            response,
            expires_at,
            created_at: now,
            access_count: 0,
            last_accessed: now,
        };
        
        let mut entries = self.entries.write()
            .map_err(|_| AIError::cache("Failed to acquire write lock"))?;
        entries.insert(key.to_string(), entry);
        
        let mut stats = self.stats.write()
            .map_err(|_| AIError::cache("Failed to acquire stats write lock"))?;
        stats.size = entries.len();
        
        Ok(())
    }
    
    fn remove(&self, key: &str) -> AIResult<bool> {
        let mut entries = self.entries.write()
            .map_err(|_| AIError::cache("Failed to acquire write lock"))?;
        let removed = entries.remove(key).is_some();
        
        let mut stats = self.stats.write()
            .map_err(|_| AIError::cache("Failed to acquire stats write lock"))?;
        stats.size = entries.len();
        
        Ok(removed)
    }
    
    fn clear(&self) -> AIResult<()> {
        let mut entries = self.entries.write()
            .map_err(|_| AIError::cache("Failed to acquire write lock"))?;
        entries.clear();
        
        let mut stats = self.stats.write()
            .map_err(|_| AIError::cache("Failed to acquire stats write lock"))?;
        *stats = CacheStats::default();
        
        Ok(())
    }
    
    fn stats(&self) -> CacheStats {
        self.stats.read()
            .map(|stats| stats.clone())
            .unwrap_or_default()
    }
    
    fn generate_key(&self, request: &AIRequest) -> String {
        use std::collections::hash_map::DefaultHasher;
        use std::hash::{Hash, Hasher};

        let mut hasher = DefaultHasher::new();
        request.feature.hash(&mut hasher);
        request.content.hash(&mut hasher);

        // Hash context manually since HashMap doesn't implement Hash
        for (key, value) in &request.context {
            key.hash(&mut hasher);
            value.hash(&mut hasher);
        }

        // Hash temperature as bits since f64 doesn't implement Hash
        if let Some(temp) = request.temperature {
            temp.to_bits().hash(&mut hasher);
        }

        request.max_tokens.hash(&mut hasher);

        format!("ai_cache_{:x}", hasher.finish())
    }
}
