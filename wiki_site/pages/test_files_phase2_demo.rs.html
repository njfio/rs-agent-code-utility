          pub fn generate_user_report(&self) -> String {
        let mut report = String::with_capacity(self.users.len() * 32);
        for (id, user) in &self.users {
            use std::fmt::Write;
            let _ = writeln!(report, "User {}: {} ({})", id, user.name, user.email);

        report
    }
