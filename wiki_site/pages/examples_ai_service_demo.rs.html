    Overview

    The examples/ai_service_demo.rs file showcases a minimal end‑to‑end demonstration of the AI service client defined in the library. It walks through creating a client, sending a simple prompt for text completion, and performing a multi‑turn chat using the OpenAI‑compatible API. The example is fully async, uses tokio for the runtime, and prints the responses to stdout, making it a handy reference for integrating the library into real applications.

    Deep Dive

      High‑level wrapper around the HTTP client and authentication for the AI service.
    • enum MessageRole

    Key APIs

    • pub async fn new(api_key: impl Into<String>) -> Result<Self, anyhow::Error>
      Creates a client; typically called once at program start. Example: let client = AiServiceClient::new(env::var("OPENAI_API_KEY")?).await?;
    • pub async fn generate_text(&self, prompt: &str, max_tokens: Option<u16>) -> Result<String, anyhow::Error>

    • pub async fn chat(&self, messages: &[Message]) -> Result<ChatResponse, anyhow::Error>

    Examples

    Simple text completion
    let client = AiServiceClient::new("sk-…").await?; let result = client.generate_text("Explain quantum entanglement in one sentence", Some(20)).await?; println!("Result: {}", result);

    Creates a client, sends a one‑shot prompt, and prints the short answer.

    Two‑turn chat demo
    let client = AiServiceClient::new(env::var("OPENAI_API_KEY")?).await?; let msgs = vec![ Message::new_system("You are a helpful assistant."), Message::new_user("What is the capital of France?"), ]; let resp = client.chat(&msgs).await?; println!("Assistant: {}", resp.choices[0].message.content);

    Shows how to seed a system prompt and a user query, then read the assistant's reply from the response structure.

    Streaming chat (advanced)
    let client = AiServiceClient::new("sk-…").await?; let msgs = vec![Message::new_user("Tell me a joke.")]; let mut stream = client.chat_stream(&msgs).await?; while let Some(chunk) = stream.next().await { print!("{}", chunk?); }

    Demonstrates the optional streaming API (available when the library is compiled with the stream feature).

    Security

    Risks
    • API key leakage if the binary is logged or the environment is exposed.
    • Prompt injection – user‑supplied content may cause the model to generate unsafe output.
    • Unencrypted HTTP traffic if the base URL is overridden to a non‑TLS endpoint.
    Mitigations
    • Load the key from a secure vault or use OS‑level secret storage; avoid printing it.
    • Validate and sanitise any user‑generated prompts before sending them to the model.
    • Enforce `https://` URLs only; the client rejects non‑TLS endpoints by default.

    println!("Starting AI Service Demo...");
    
    println!("🤖 AI Service Demo");
    println!("==================");
    
    // Demo 1: Load configuration from file
    println!("\n📁 Loading configuration from file...");
    match load_config_from_file().await {
        Ok(_) => println!("✅ Configuration loaded successfully"),
        Err(e) => println!("❌ Failed to load config: {}", e),
    }
    
    // Demo 2: Create configuration programmatically
    println!("\n⚙️  Creating configuration programmatically...");

    test_ai_features(&service).await?;
    
    // Demo 4: Show cache statistics
    println!("\n📊 Cache statistics:");
    if let Some(stats) = service.cache_stats() {
        println!("  Hits: {}, Misses: {}, Hit Rate: {:.2}%", 
                 stats.hits, stats.misses, stats.hit_rate * 100.0);
    } else {
        println!("  Cache disabled");
    }
    
    // Demo 5: Validate provider connections
    println!("\n🔗 Validating provider connections...");
    let validation_results = service.validate_connections().await;
    for (provider, result) in validation_results {
        match result {
            Ok(_) => println!("  {:?}: ✅ Connected", provider),
            Err(e) => println!("  {:?}: ❌ Error: {}", provider, e),
        }
    }
    
    println!("\n🎉 Demo completed!");
    Ok(())
}

      async fn load_config_from_file() -> Result<AIService, Box<dyn std::error::Error>> {
    // Try to load from JSON file first, then YAML
    let config = if std::path::Path::new("ai_config.json").exists() {
        AIServiceConfig::from_file("ai_config.json")?
    } else if std::path::Path::new("ai_config.yaml").exists() {
        AIServiceConfig::from_file("ai_config.yaml")?
    } else {
        println!("  No config file found, using defaults");
        AIServiceConfig::default()
    };
    
    let service = AIServiceBuilder::new()
        .with_config(config)
        .with_mock_providers(true) // Use mock providers for demo
        .build()
        .await?;
    
    Ok(service)
}

      async fn create_programmatic_config() -> Result<AIService, Box<dyn std::error::Error>> {
    // Create a simple configuration programmatically
    let mut config = AIServiceConfig::default();
    config.default_provider = AIProvider::OpenAI;
    
    // Configure OpenAI provider
    let openai_config = ProviderConfig {
        enabled: true,
        api_key: Some("demo-key".to_string()),
        base_url: Some("https://api.openai.com/v1".to_string()),
        organization: None,
        models: vec![
            ModelConfig {
                name: "gpt-4".to_string(),
                context_length: 8192,
                max_tokens: 4096,
                supports_streaming: true,
                cost_per_token: Some(0.00003),
                supported_features: vec![
                    AIFeature::CodeExplanation,
                    AIFeature::SecurityAnalysis,
                    AIFeature::RefactoringSuggestions,
                ],
            }
        ],
        default_model: "gpt-4".to_string(),
        timeout: Duration::from_secs(30),
        rate_limit: RateLimitConfig {
            requests_per_minute: 60,
            tokens_per_minute: 100000,
            burst_size: 10,
        },
        retry: RetryConfig {
            max_retries: 3,
            initial_delay: Duration::from_secs(1),
            max_delay: Duration::from_secs(30),
            backoff_multiplier: 2.0,
        },
    };
    
    config.providers.insert(AIProvider::OpenAI, openai_config);
    
    let service = AIServiceBuilder::new()
        .with_config(config)
        .with_mock_providers(true) // Use mock providers for demo
        .build()
        .await?;
    
    Ok(service)
}
