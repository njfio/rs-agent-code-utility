    Overview

    The src/analysis_utils.rs module provides a collection of helper functions and data structures for performing static and runtime analysis on Rust codebases. It includes utilities for token stream inspection, AST traversal, metric computation (e.g., cyclomatic complexity, LOC), and pattern matching on syntax nodes. The module is deliberately lightweight and has no external dependencies beyond syn, quote, and the standard library, making it suitable for integration into linters, code generators, and IDE plugins.

    Examples

    Counting functions and total LOC in a crate
    use std::fs; use syn::{self, File}; use analysis_utils::{visit_functions, Metric};

fn analyze_path(path: &std::path::Path) -> Result<Metric, analysis_utils::AnalysisError> {
    let src = fs::read_to_string(path)?;
    let ast: File = syn::parse_file(&src)?;
    let mut metric = Metric::new();
    metric.lines_of_code = src.lines().count();
    visit_functions(&ast, |func| {
        metric.function_count += 1;
        metric.cyclomatic_complexity += analysis_utils::compute_cyclomatic_complexity(&func.block);
    })?;
    Ok(metric)
}

fn main() -> Result<(), analysis_utils::AnalysisError> {
    let mut total = Metric::new();
    for entry in walkdir::WalkDir::new("src") {
        let entry = entry?;
        if entry.path().extension().and_then(|s| s.to_str()) == Some("rs") {
            let m = analyze_path(entry.path())?;
            total.merge(&m);
        }
    }
    println!("Total functions: {}", total.function_count);
    println!("Total LOC: {}", total.lines_of_code);
    println!("Avg CC: {:.2}", total.cyclomatic_complexity as f64 / total.function_count as f64);
    Ok(())
}

    The example walks a directory, parses each .rs file, uses visit_functions to count functions and compute cyclomatic complexity, and aggregates the results in a Metric instance.

    Security

    Risks
    • Parsing untrusted source files can trigger panics in older versions of `syn` if the input is malformed. The module catches `syn::Error` and wraps it in `AnalysisError`, but a panic before that point is possible.
    • If the analysis results are later used to generate code (e.g., via `quote!`), ensure proper sanitisation of identifiers to avoid injection attacks.
    Mitigations
    • Run analysis in a sandboxed thread with a panic hook that converts panics to `AnalysisError`.
    • Validate all identifiers with `syn::Ident::new` before emitting them in generated code.

          pub fn calculate_cyclomatic_complexity(tree: &SyntaxTree, language: &str) -> f64 {
        let mut complexity = 1.0; // Base complexity
        let control_patterns = LanguageParser::get_control_flow_patterns(language);
        
        // Count control flow nodes
        for pattern in &control_patterns {
            let nodes = tree.find_nodes_by_kind(pattern);
            complexity += nodes.len() as f64;
        }
        
        complexity
    }

          pub fn calculate_function_complexity(func_node: &Node, language: &str) -> f64 {
        let mut complexity = 1.0; // Base complexity
        let control_patterns = LanguageParser::get_control_flow_patterns(language);

        // Count control flow statements within the function
        for pattern in &control_patterns {
            let child_nodes = func_node.find_descendants(|node| node.kind() == *pattern);
            complexity += child_nodes.len() as f64;
        }

        complexity
    }

          pub fn calculate_method_complexity_score(symbol: &Symbol, line_count: usize) -> f64 {
        let mut score = 0.0;

        // Base score from line count
        score += (line_count as f64 / 30.0).min(1.0) * 0.5;

        // Add complexity based on symbol name patterns (heuristic)
        if symbol.name.contains("_and_") || symbol.name.contains("_or_") {
            score += 0.2;
        }

        // Add complexity for long names (often indicate complex functionality)
        if symbol.name.len() > 20 {
            score += 0.1;
        }

        score.min(1.0)
    }

          pub fn count_nested_loops(content: &str, language: &str) -> usize {
        let loop_patterns = LanguageParser::get_loop_patterns(language);
        let lines: Vec<&str> = content.lines().collect();
        let mut max_depth = 0;
        let mut current_depth = 0;

        for line in &lines {
            let trimmed = line.trim();
            
            // Check for loop start patterns
            for pattern in &loop_patterns {
                if trimmed.contains(pattern.replace("_", " ").as_str()) {
                    current_depth += 1;
                    max_depth = max_depth.max(current_depth);
                    break;
                }
            }

            // Simple brace counting for depth tracking
            if trimmed == "}" && current_depth > 0 {
                current_depth -= 1;
            }
        }

        max_depth
    }

        let mut in_loop = false;

        for line in &lines {
            let trimmed = line.trim();
            
            if trimmed.contains("for ") || trimmed.contains("while ") {
                in_loop = true;
            }
            
            if in_loop && (trimmed.contains(" + ") || trimmed.contains("+=")) 
                && (trimmed.contains("String") || trimmed.contains("&str")) {
                return true;
            }
            
            if trimmed == "}" {
                in_loop = false;
            }
        }

        false
    }

          fn calculate_name_similarity(name1: &str, name2: &str) -> f64 {
        // Simple Levenshtein-like similarity
        let len1 = name1.len();
        let len2 = name2.len();
        
        if len1 == 0 || len2 == 0 {
            return 0.0;
        }
        
        let common_chars = name1.chars()
            .filter(|c| name2.contains(*c))
            .count();
            
        common_chars as f64 / len1.max(len2) as f64
    }

          fn calculate_size_similarity(symbol1: &Symbol, symbol2: &Symbol) -> f64 {
        let size1 = SymbolFilter::calculate_line_count(symbol1) as f64;
        let size2 = SymbolFilter::calculate_line_count(symbol2) as f64;
        
        let diff = (size1 - size2).abs();
        let max_size = size1.max(size2);
        
        if max_size == 0.0 {
            1.0
        } else {
            1.0 - (diff / max_size).min(1.0)
        }
    }

      impl ComplexityCalculator {
    /// Calculate cyclomatic complexity from AST
    pub fn calculate_cyclomatic_complexity(tree: &SyntaxTree, language: &str) -> f64 {
        let mut complexity = 1.0; // Base complexity
        let control_patterns = LanguageParser::get_control_flow_patterns(language);
        
        // Count control flow nodes
        for pattern in &control_patterns {
            let nodes = tree.find_nodes_by_kind(pattern);
            complexity += nodes.len() as f64;
        }
        
        complexity
    }

    /// Calculate function complexity from node
    pub fn calculate_function_complexity(func_node: &Node, language: &str) -> f64 {
        let mut complexity = 1.0; // Base complexity
        let control_patterns = LanguageParser::get_control_flow_patterns(language);

        // Count control flow statements within the function
        for pattern in &control_patterns {
            let child_nodes = func_node.find_descendants(|node| node.kind() == *pattern);
            complexity += child_nodes.len() as f64;
        }

        complexity
    }

    /// Calculate method complexity score based on multiple factors
    pub fn calculate_method_complexity_score(symbol: &Symbol, line_count: usize) -> f64 {
        let mut score = 0.0;

        // Base score from line count
        score += (line_count as f64 / 30.0).min(1.0) * 0.5;

        // Add complexity based on symbol name patterns (heuristic)
        if symbol.name.contains("_and_") || symbol.name.contains("_or_") {
            score += 0.2;
        }

        // Add complexity for long names (often indicate complex functionality)
        if symbol.name.len() > 20 {
            score += 0.1;
        }

        score.min(1.0)
    }
}

    /// Count nested loops in content
    pub fn count_nested_loops(content: &str, language: &str) -> usize {
        let loop_patterns = LanguageParser::get_loop_patterns(language);
        let lines: Vec<&str> = content.lines().collect();
        let mut max_depth = 0;
        let mut current_depth = 0;

        for line in &lines {
            let trimmed = line.trim();
            
            // Check for loop start patterns
            for pattern in &loop_patterns {
                if trimmed.contains(pattern.replace("_", " ").as_str()) {
                    current_depth += 1;
                    max_depth = max_depth.max(current_depth);
                    break;
                }
            }

            // Simple brace counting for depth tracking
            if trimmed == "}" && current_depth > 0 {
                current_depth -= 1;
            }
        }

        max_depth
    }

    /// Detect string concatenation in loops
    pub fn detect_string_concatenation_in_loops(content: &str) -> bool {
        let lines: Vec<&str> = content.lines().collect();
        let mut in_loop = false;

        for line in &lines {
            let trimmed = line.trim();
            
            if trimmed.contains("for ") || trimmed.contains("while ") {
                in_loop = true;
            }
            
            if in_loop && (trimmed.contains(" + ") || trimmed.contains("+=")) 
                && (trimmed.contains("String") || trimmed.contains("&str")) {
                return true;
            }
            
            if trimmed == "}" {
                in_loop = false;
            }
        }

        false
    }

    /// Calculate function similarity based on name and structure
    pub fn calculate_function_similarity(symbol1: &Symbol, symbol2: &Symbol) -> f64 {
        // Simple similarity based on name patterns and line counts
        let name_similarity = Self::calculate_name_similarity(&symbol1.name, &symbol2.name);
        let size_similarity = Self::calculate_size_similarity(symbol1, symbol2);
        
        (name_similarity + size_similarity) / 2.0
    }

    fn calculate_name_similarity(name1: &str, name2: &str) -> f64 {
        // Simple Levenshtein-like similarity
        let len1 = name1.len();
        let len2 = name2.len();
        
        if len1 == 0 || len2 == 0 {
            return 0.0;
        }
        
        let common_chars = name1.chars()
            .filter(|c| name2.contains(*c))
            .count();
            
        common_chars as f64 / len1.max(len2) as f64
    }

    fn calculate_size_similarity(symbol1: &Symbol, symbol2: &Symbol) -> f64 {
        let size1 = SymbolFilter::calculate_line_count(symbol1) as f64;
        let size2 = SymbolFilter::calculate_line_count(symbol2) as f64;
        
        let diff = (size1 - size2).abs();
        let max_size = size1.max(size2);
        
        if max_size == 0.0 {
            1.0
        } else {
            1.0 - (diff / max_size).min(1.0)
        }
    }
