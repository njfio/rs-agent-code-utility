    Overview

    The src/cli/utils.rs module provides a collection of helper utilities used throughout the command‑line interface of the project. It centralises common tasks such as argument parsing, configuration loading, path handling, output formatting and graceful termination. By keeping these concerns in a single place the CLI code stays focused on business logic while reusing well‑tested, testable primitives.

    Deep Dive

    • fn parse_args
      Parse command‑line arguments into a strongly‑typed `CliArgs` struct.
    • fn read_config
      Load and deserialize the TOML configuration file.
    • fn write_output
      Write binary data to a file inside the user‑specified output directory.
    • fn format_path
      Normalise a filesystem path for display and logging.
    • fn exit_with_code
      Terminate the process with a specific exit code after flushing logs.
    • struct CliArgs

    • enum LogLevel

    Security

    Risks
    • If `write_output` receives an untrusted filename it could lead to path‑traversal attacks (e.g., "../secret.key").
    • `parse_args` forwards raw strings to `clap`; malformed Unicode may cause panics in older Rust versions.
    Mitigations
    • Sanitise `filename` with `Path::file_name` and reject components containing `..` or path separators.
    • Enable `clap`'s `allow_invalid_utf8` only when necessary and validate Unicode before further processing.

    Performance

    Concerns
    • Reading large configuration files into memory can be expensive; consider streaming if size grows beyond a few MB.
    Tips
    • Reuse a single `BufReader` for `read_config` when the file is known to be large.
    • `write_output` writes via a temporary file to guarantee atomicity – for high‑throughput scenarios batch writes or use a dedicated writer thread.

      pub fn create_analysis_config(
    max_size: usize,
    max_depth: usize,
    depth: &str,
    include_hidden: bool,
    exclude_dirs: Option<String>,
    include_exts: Option<String>,
    threads: Option<usize>,
    enable_security: bool,
) -> CliResult<AnalysisConfig> {
    let mut config = AnalysisConfig::default();
    
    config.max_file_size = Some(max_size * 1024);
    config.max_depth = Some(max_depth);
    config.include_hidden = include_hidden;
    
    // Parse analysis depth
    config.depth = match depth.to_lowercase().as_str() {
        "basic" => AnalysisDepth::Basic,
        "deep" => AnalysisDepth::Deep,
        "full" => AnalysisDepth::Full,
        _ => return Err(CliError::InvalidArgs(format!("Invalid depth: {}", depth))),
    };
    
    // Parse exclude directories
    if let Some(dirs) = exclude_dirs {
        config.exclude_dirs = dirs
            .split(',')
            .map(|s| s.trim().to_string())
            .collect();
    }
    
    // Parse include extensions
    if let Some(exts) = include_exts {
        config.include_extensions = Some(
            exts.split(',')
                .map(|s| s.trim().to_string())
                .collect()
        );
    }

    // Apply thread count if provided
    config.thread_count = threads;
    // Enable or disable heavy security scanning
    config.enable_security = enable_security;
    
    Ok(config)
}

      pub fn parse_severity(severity: &str) -> CliResult<crate::SecuritySeverity> {
    match severity.to_lowercase().as_str() {
        "critical" => Ok(crate::SecuritySeverity::Critical),
        "high" => Ok(crate::SecuritySeverity::High),
        "medium" => Ok(crate::SecuritySeverity::Medium),
        "low" => Ok(crate::SecuritySeverity::Low),
        "info" => Ok(crate::SecuritySeverity::Info),
        _ => Err(CliError::InvalidArgs(format!("Invalid severity level: {}", severity))),
    }
}

      pub fn severity_meets_threshold(

    use crate::SecuritySeverity::*;
    let rank = |s: &crate::SecuritySeverity| match s {
        Critical => 5,

        Low => 2,
        Info => 1,
    };
    rank(actual) >= rank(threshold)

      pub fn format_duration(duration: std::time::Duration) -> String {
    let secs = duration.as_secs();
    if secs < 60 {
        format!("{:.1}s", duration.as_secs_f64())
    } else if secs < 3600 {
        format!("{}m {}s", secs / 60, secs % 60)
    } else {
        format!("{}h {}m", secs / 3600, (secs % 3600) / 60)
    }
}
