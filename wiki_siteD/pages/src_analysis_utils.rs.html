    Overview

    The analysis_utils module provides a collection of helper functions and data structures for performing statistical and performance analysis on data streams generated by the application. It includes utilities for parsing raw logs, computing common metrics (mean, median, percentiles, throughput), aggregating results across threads, and generating human‑readable summaries. The module is deliberately lightweight and has no external dependencies beyond the Rust standard library and serde for optional JSON serialization.

    Deep Dive

    • struct AnalysisResult
      Container for a set of computed metrics for a single analysis run.
    • enum Metric
      Enumerates the supported metric types.
    • fn parse_log_line
      Parses a single line of the application log into a numeric value.
    • fn compute_metric
      Computes a selected metric over a slice of samples.
    • fn aggregate_results
      Reduces a vector of `AnalysisResult` into a single combined result.

    Basic end‑to‑end analysis
    use analysis_utils::{parse_log_line, compute_metric, Metric, AnalysisResult, aggregate_results};

fn main() -> Result<(), Box<dyn std::error::Error>> {
    // Simulate reading a log file line‑by‑line
    let log = "12.3\n15.7\n9.8\n13.4\n";
    let mut samples = Vec::new();
    for line in log.lines() {
        let v = parse_log_line(line)?;
        samples.push(v);
    }

    // Compute a few metrics
    let mean = compute_metric(&samples, Metric::Mean)?;
    let p95 = compute_metric(&samples, Metric::Percentile(95.0))?;

    // Pack into an AnalysisResult
    let result = AnalysisResult::new(samples.len(), samples.iter().sum(),
                                    compute_metric(&samples, Metric::Median)?,
                                    p95,
                                    compute_metric(&samples, Metric::Percentile(99.0))?);
    println!("Mean: {:.2}, 95th‑pct: {:.2}", mean, p95);

    // If we had multiple shards we could aggregate them:
    let combined = aggregate_results(&[result.clone(), result]);
    println!("Combined mean: {:.2}", combined.mean);
    Ok(())
}

    The example demonstrates the typical workflow: parse raw log lines, collect numeric samples, compute several metrics, wrap them in an AnalysisResult, and finally aggregate multiple results (e.g., from parallel workers).

fn parallel_reduce(partials: Vec<AnalysisResult>) -> AnalysisResult {
    // Rayon will split the vector and merge in parallel using the internal `merge` implementation.
    partials.par_iter().cloned().reduce(|| AnalysisResult::default(), |a, b| a.merge(&b))

          pub fn count_nested_loops(content: &str, language: &str) -> usize {
        let loop_patterns = LanguageParser::get_loop_patterns(language);
        let lines: Vec<&str> = content.lines().collect();
        let mut max_depth = 0;
        let mut current_depth = 0;

        for line in &lines {
            let trimmed = line.trim();
            
            // Check for loop start patterns
            for pattern in &loop_patterns {
                if trimmed.contains(pattern.replace("_", " ").as_str()) {
                    current_depth += 1;
                    max_depth = max_depth.max(current_depth);
                    break;
                }
            }

            // Simple brace counting for depth tracking
            if trimmed == "}" && current_depth > 0 {
                current_depth -= 1;
            }
        }

        max_depth
    }

          fn calculate_name_similarity(name1: &str, name2: &str) -> f64 {
        // Simple Levenshtein-like similarity
        let len1 = name1.len();
        let len2 = name2.len();
        
        if len1 == 0 || len2 == 0 {
            return 0.0;
        }
        
        let common_chars = name1.chars()
            .filter(|c| name2.contains(*c))
            .count();
            
        common_chars as f64 / len1.max(len2) as f64
    }

    }

    /// Count nested loops in content
    pub fn count_nested_loops(content: &str, language: &str) -> usize {
        let loop_patterns = LanguageParser::get_loop_patterns(language);
        let lines: Vec<&str> = content.lines().collect();
        let mut max_depth = 0;
        let mut current_depth = 0;

        for line in &lines {
            let trimmed = line.trim();
            
            // Check for loop start patterns
            for pattern in &loop_patterns {
                if trimmed.contains(pattern.replace("_", " ").as_str()) {
                    current_depth += 1;
                    max_depth = max_depth.max(current_depth);
                    break;
                }
            }

            // Simple brace counting for depth tracking
            if trimmed == "}" && current_depth > 0 {
                current_depth -= 1;
            }
        }

        max_depth
    }

    /// Detect string concatenation in loops
    pub fn detect_string_concatenation_in_loops(content: &str) -> bool {
        let lines: Vec<&str> = content.lines().collect();
        let mut in_loop = false;

        for line in &lines {
            let trimmed = line.trim();
            
            if trimmed.contains("for ") || trimmed.contains("while ") {
                in_loop = true;
            }
            
            if in_loop && (trimmed.contains(" + ") || trimmed.contains("+=")) 
                && (trimmed.contains("String") || trimmed.contains("&str")) {
                return true;
            }
            
            if trimmed == "}" {
                in_loop = false;
            }
        }

        false
    }

    fn calculate_name_similarity(name1: &str, name2: &str) -> f64 {
        // Simple Levenshtein-like similarity
        let len1 = name1.len();
        let len2 = name2.len();
        
        if len1 == 0 || len2 == 0 {
            return 0.0;
        }
        
        let common_chars = name1.chars()
            .filter(|c| name2.contains(*c))
            .count();
            
        common_chars as f64 / len1.max(len2) as f64
    }

    fn calculate_size_similarity(symbol1: &Symbol, symbol2: &Symbol) -> f64 {
        let size1 = SymbolFilter::calculate_line_count(symbol1) as f64;
        let size2 = SymbolFilter::calculate_line_count(symbol2) as f64;
        
        let diff = (size1 - size2).abs();
        let max_size = size1.max(size2);
        
        if max_size == 0.0 {
            1.0
        } else {
            1.0 - (diff / max_size).min(1.0)
        }
    }
