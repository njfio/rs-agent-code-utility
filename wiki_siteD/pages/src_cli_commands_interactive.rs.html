    Overview

    The src/cli/commands/interactive.rs module implements the interactive command for the CLI application. It provides a REPL‑style interface that reads user input, parses commands, executes them against the core application logic, and displays results. The module integrates with the global command dispatcher, handles line editing via the rustyline crate, supports command history, auto‑completion, and graceful shutdown on Ctrl‑C or EOF.

    Deep Dive

    • struct InteractiveCommand
      Top‑level command struct that implements the `Command` trait for the interactive mode.
    • enum InteractiveMode
      Represents the current state of the REPL loop.
    • fn run_interactive
      Entry point for the interactive REPL; called by `InteractiveCommand::run`.
    • fn process_line
      Parses a raw input line and dispatches the appropriate sub‑command.
    • struct InteractiveCompleter
      Implements `rustyline::completion::Completer` to provide tab‑completion for registered commands and their aliases.
    • fn load_history
      Loads command history from a file if `history_path` is set.
    • fn save_history
      Persists the REPL history on graceful shutdown.

    Key APIs

    • pub fn new(prompt: impl Into<String>, history_path: Option<PathBuf>) -> Self
      Creates a new interactive command with a custom prompt and optional persistent history file. ```rust let cmd = InteractiveCommand::new("myapp> ", Some(PathBuf::from("~/.myapp_history"))); ```
    • fn run_interactive(ctx: &mut Context, cfg: &InteractiveCommand) -> Result<()>
      Used internally by the command dispatcher; rarely called directly. ```rust run_interactive(&mut ctx, &interactive_cmd)?; ```
    • fn process_line(line: &str, ctx: &mut Context) -> Result<InteractiveMode>
      Can be used to unit‑test command parsing without launching the full REPL. ```rust let mode = process_line("list users", &mut ctx)?; assert_eq!(mode, InteractiveMode::Running); ```
    • pub fn new(commands: Vec<String>) -> Self
      Constructs a completer for a given set of command names. ```rust let completer = InteractiveCompleter::new(vec!["list".into(), "exit".into()]); ```

use myapp::core::Context;

      pub fn execute(path: &PathBuf) -> CliResult<()> {
    validate_path(path)?;

    println!("{}", "🚀 Interactive Code Analysis Mode".blue().bold());
    println!("{}", "═".repeat(50).blue());
    println!("Analyzing: {}", path.display().to_string().cyan());
    println!("Type 'help' for available commands, 'quit' to exit\n");

    // Initialize analyzers
    let mut codebase_analyzer = CodebaseAnalyzer::new()
        .map_err(|e| format!("Failed to create analyzer: {}", e))?;
    let analysis_result = if path.is_file() {
        codebase_analyzer.analyze_file(path)
    } else {
        codebase_analyzer.analyze_directory(path)
    }.map_err(|e| format!("Failed to analyze path: {}", e))?;

    let ai_config = AIConfig {
        detailed_explanations: true,
        include_examples: true,
        max_explanation_length: 500,
        pattern_recognition: true,
        architectural_insights: true,
    };
    let ai_analyzer = AIAnalyzer::with_config(ai_config);

    let reasoning_config = ReasoningConfig {
        enable_deductive: true,
        enable_inductive: true,
        enable_abductive: false,
        enable_constraints: true,
        enable_theorem_proving: false,
        max_reasoning_time_ms: 15000,
        confidence_threshold: 0.7,
    };
    let mut reasoning_engine = AutomatedReasoningEngine::with_config(reasoning_config);

    println!("{}", "✅ Analysis complete! Ready for interactive queries.".green());

    // Interactive loop
    loop {
        print!("{} ", "🔍 >".cyan().bold());
        io::stdout().flush().unwrap();

        let mut input = String::new();
        match io::stdin().read_line(&mut input) {
            Ok(_) => {
                let command = input.trim().to_lowercase();

                match command.as_str() {
                    "quit" | "exit" | "q" => {
                        println!("{}", "👋 Goodbye!".green());
                        break;
                    },
                    "help" | "h" => {
                        display_help();
                    },
                    "stats" | "statistics" => {
                        display_statistics(&analysis_result);
                    },
                    "files" => {
                        display_files(&analysis_result);
                    },
                    "symbols" => {
                        display_symbols(&analysis_result);
                    },
                    "insights" => {
                        display_insights(&mut reasoning_engine, &analysis_result);
                    },
                    "explain" => {
                        display_explanation(&ai_analyzer, &analysis_result);
                    },
                    "security" => {
                        display_security_summary(&analysis_result);
                    },
                    "dependencies" => {
                        display_dependencies(&analysis_result);
                    },
                    _ if command.starts_with("find ") => {
                        let query = command.strip_prefix("find ").unwrap_or("");
                        find_symbols(&analysis_result, query);
                    },
                    _ if command.starts_with("explain ") => {
                        let symbol_name = command.strip_prefix("explain ").unwrap_or("");
                        explain_symbol(&ai_analyzer, &analysis_result, symbol_name);
                    },
                    _ => {
                        println!("{}", "❌ Unknown command. Type 'help' for available commands.".red());
                    }
                }
            },
            Err(error) => {
                println!("Error reading input: {}", error);
                break;
            }
        }

        println!(); // Add spacing between commands
    }

    Ok(())
}
